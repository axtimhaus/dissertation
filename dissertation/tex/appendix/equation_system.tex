\section{Components of the Lagrangian Gradient}\label{sec:components-of-the-lagrangian-gradient}

The following table displays all components of the Lagrangian gradient to be solved for the application of the \gls{TEP}.
Node symbols $n$ or contact symbols $c$ in parentheses behind another symbol mean, that this symbol is local to the node or contact, thus there exist one distinct value for each node or contact.
$n_{\Upper}$ denotes the upper neighbor of the currently regarded node, as well as $n_{\Lower}$ the lower.
The equations are all formulated in terms of fluxes from one node to the upper neighbor $\Flux_{\Upper}$, since $\Flux_{\Lower}(n) = -\Flux_{\Upper}(n_{\Lower})$ must be always fulfilled.
This saves one unknown per node.
Partial derivatives, that have been left standing symbolically, are given in the text part of this document, especially in \cref{sec:surface-evolution,sec:contact-conditions}.
They depend only on the current geometry and constant material parameters (especially the local interface energy).

\begin{longtblr}[
        label=none,
        entry=none,
        note{*}={Only if the node is in parent position.}
    ]{
        colspec={lX[c]},
        measure=vbox
    }
    \toprule
    \SetCell[c=2]{l} for each non-contact node $n$ \\
    \midrule
    $\LagrangeFunction_{\dot\Shift_{\Normal}(n)}$ &
    \begin{equation*}
        0
        = -\frac{\partial\GibbsEnergy}{\partial\Shift_{\Normal}(n)} \left( 1 + \LagrangeParameter_{\Dissipation} \right)
        + \frac{\partial{\Volume}(n)}{\partial\Shift_{\Normal}(n)} \LagrangeParameter_{\Volume}(n)
    \end{equation*} \\

    $\LagrangeFunction_{\Flux_{\Upper}(n)}$ &
    \begin{equation*}
        0
        = -\frac{2 \GasConstant \Temperature}{\MolarVolume \VacancyConcentration^{\Standard}} \frac{\SurfaceDistance_{\Upper}(n) \Flux_{\Upper}(n)}{\DiffusionCoefficient_{\Upper}(n)} \LagrangeParameter_{\Dissipation}
        - \LagrangeParameter_{\Volume}(n)
        + \LagrangeParameter_{\Volume}(n_{\Upper})
    \end{equation*} \\

    $\LagrangeFunction_{\LagrangeParameter_{\Volume}(n)}$ &
    \begin{equation*}
        0
        = \frac{\partial{\Volume}(n)}{\partial\Shift_{\Normal}} \dot\Shift_{\Normal}(n)
        - \Flux_{\Upper}(n)
        + \Flux_{\Upper}(n_{\Lower})
    \end{equation*} \\

    \midrule
    \SetCell[c=2]{l} for each contact node $n$ as part of contact $c$ \\
    \midrule
    $\LagrangeFunction_{\dot\Shift_{\Normal}(n)}$ &
    \begin{equation*}
        0
        = -\frac{\partial\GibbsEnergy}{\partial\Shift_{\Normal}(n)} \left( 1 + \LagrangeParameter_{\Dissipation} \right)
        + \frac{\partial{\Volume}(n)}{\partial\Shift_{\Normal}(n)} \LagrangeParameter_{\Volume}(n)
        - \frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Normal}(n)} \LagrangeParameter_{\Radius_{\Contact}}(n)
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Normal}(n)} \LagrangeParameter_{\Angle_{\Contact}}(n)
    \end{equation*} \\

    $\LagrangeFunction_{\dot\Shift_{\Tangential}(n)}$ &
    \begin{equation*}
        0
        = -\frac{\partial\GibbsEnergy}{\partial\Shift_{\Tangential}(n)} \left( 1 + \LagrangeParameter_{\Dissipation} \right)
        + \frac{\partial{\Volume}(n)}{\partial\Shift_{\Tangential}(n)} \LagrangeParameter_{\Volume}(n)
        - \frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)} \LagrangeParameter_{\Radius_{\Contact}}(n)
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)} \LagrangeParameter_{\Angle_{\Contact}}(n)
    \end{equation*} \\

    $\LagrangeFunction_{\Flux_{\Upper}(n)}$ & same as above \\

    $\LagrangeFunction_{\LagrangeParameter_{\Volume}(n)}$ &
    \begin{equation*}
        0
        = \frac{\partial{\Volume}(n)}{\partial\Shift_{\Normal}} \dot\Shift_{\Normal}(n)
        + \frac{\partial{\Volume}(n)}{\partial\Shift_{\Tangential}} \dot\Shift_{\Tangential}(n)
        - \Flux_{\Upper}(n)
        + \Flux_{\Upper}(n_{\Lower})
    \end{equation*} \\

    $\LagrangeFunction_{\LagrangeParameter_{\Radius_{\Contact}}(n)}$\TblrNote{*} &
    \begin{multline*}
        0
        = \dot\Radius_{\Contact}(c)
        - \frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Normal}(n)} \left( \dot\Shift_{\Normal}\Regarding\Parent(n) + \dot\Shift_{\Normal}\Regarding\Child(n) \right)
        - \frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)} \left( \dot\Shift_{\Tangential}\Regarding\Parent(n) + \dot\Shift_{\Tangential}\Regarding\Child(n) \right) \\
        - 2 \frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\RotationAngle}(n)} \Radius\Regarding\Child(n) \sin \frac{\dot\RotationAngle_{\Child}(c)}{2}
    \end{multline*} \\

    $\LagrangeFunction_{\LagrangeParameter_{\Angle_{\Contact}}(n)}$\TblrNote{*} &
    \begin{multline*}
        0
        = \dot\Angle_{\Contact}(c)
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Normal}(n)} \left( \dot\Shift_{\Normal}\Regarding\Parent(n) + \dot\Shift_{\Normal}\Regarding\Child(n) \right)
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)} \left( \dot\Shift_{\Tangential}\Regarding\Parent(n) + \dot\Shift_{\Tangential}\Regarding\Child(n) \right) \\
        - 2 \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\RotationAngle}(n)} \Radius\Regarding\Child(n) \sin \frac{\dot\RotationAngle_{\Child}(c)}{2}
    \end{multline*} \\

    \midrule
    \SetCell[c=2]{l} for each contact $c$ with
    \begin{itemize}
        \item the therein involved nodes of the parent $\Nodes\Regarding\Parent(c)$,
        \item the rings this contact is part of $\Rings(c)$,
        \item the parent particle of this contact $p$
        \item $\delta(c)$ as a sign function that is $1$ at the primary path and $-1$ at the secondary path
    \end{itemize} \\
    \midrule
    $\LagrangeFunction_{\dot\Radius_{\Contact}(c)}$ &
    \begin{equation*}
        0
        = \sum^{\Nodes\Regarding\Parent(c)}_n \LagrangeParameter_{\Radius_{\Contact}}(n)
        + \sum_r^{\Rings(c)} \delta(c) \left[
            \cos \left( \RotationAngle(p) + \Angle_{\Contact}(c) \right) \LagrangeParameter_{\Ring\X}(r)
            + \sin \left( \RotationAngle(p) + \Angle_{\Contact}(c) \right) \LagrangeParameter_{\Ring\Y}(r)
        \right]
    \end{equation*} \\

    $\LagrangeFunction_{\dot\Angle_{\Contact}(c)}$ &
    \begin{multline*}
        0
        = \sum^{\Nodes\Regarding\Parent(c)}_n \LagrangeParameter_{\Angle_{\Contact}}(n)\\
        + \sum_r^{\Rings(c)} \delta(c) \Radius_{\Contact}(c) \left[
            -\sin \left( \RotationAngle(p) + \Angle_{\Contact}(c) \right) \LagrangeParameter_{\Ring\X}(r)
            + \cos \left( \RotationAngle(p) + \Angle_{\Contact}(c) \right) \LagrangeParameter_{\Ring\Y}(r)
        \right]
    \end{multline*} \\

    $\LagrangeFunction_{\dot\RotationAngle_{\Child}(c)}$ &
    \begin{multline*}
        0
        = \sum^{\Nodes\Regarding\Parent(c)}_n \Biggl[
            - \Radius\Regarding\Child(n) \sin \left( \Angle\Regarding\Child(n) - \Angle_{\Contact}\Regarding\Child(c) \right) \LagrangeParameter_{\Radius_{\Contact}}(n) \\
            + \frac{\Radius\Regarding\Child(n)}{\Radius_{\Contact}(c)} \cos \left( \Angle\Regarding\Child(n) - \Angle_{\Contact}\Regarding\Child(c)  \right) \LagrangeParameter_{\Angle_{\Contact}}(n)
        \Biggr]
    \end{multline*} \\

    \midrule
    \SetCell[c=2]{l} for each ring $r$ with
    \begin{itemize}
        \item the therein involved contacts $\Contacts(r)$,
        \item the parent particle of a contact $p(c)$
        \item $\delta(c)$ as a sign function that is $1$ at the primary path and $-1$ at the secondary path
    \end{itemize} \\
    \midrule

    $\LagrangeFunction_{\LagrangeParameter_{\Ring\X}(r)}$ &
    \begin{equation*}
        0
        = \sum_c^{\Contacts(r)} \delta(c)  \left[
            \cos \left( \RotationAngle(p(c)) + \Angle_{\Contact}(c) \right) \dot\Radius_{\Contact}(c)
            - \Radius_{\Contact}(c) \sin \left( \RotationAngle(p(c)) + \Angle_{\Contact}(c) \right) \dot\Angle_{\Contact}(c)
        \right]
    \end{equation*} \\

    $\LagrangeFunction_{\LagrangeParameter_{\Ring\Y}(r)}$ &
    \begin{equation*}
        0
        = \sum_c^{\Contacts(r)} \delta(c)  \left[
            \sin \left( \RotationAngle(p(c)) + \Angle_{\Contact}(c) \right) \dot\Radius_{\Contact}(c)
            + \Radius_{\Contact}(c) \cos \left( \RotationAngle(p(c)) + \Angle_{\Contact}(c) \right) \dot\Angle_{\Contact}(c)
        \right]
    \end{equation*} \\

    \midrule
    \SetCell[c=2]{l} just once for the set of all nodes $\Nodes$ \\
    \midrule
    $\LagrangeFunction_{\LagrangeParameter_{\Dissipation}}$ &
    \begin{equation*}
        0
        = \sum^{\Nodes}_n \left[
            \frac{\partial\GibbsEnergy}{\partial\Shift_{\Normal}(n)} \dot\Shift_{\Normal}(n)
            + \frac{\partial\GibbsEnergy}{\partial\Shift_{\Tangential}(n)} \dot\Shift_{\Tangential}(n)
            + \frac{\GasConstant\Temperature\SurfaceDistance_{\Upper}(n)}{\MolarVolume\VacancyConcentration^{\Standard} \DiffusionCoefficient_{\Upper}}\Flux_{\Upper}(n)^2
        \right]
    \end{equation*} \\
    \bottomrule
\end{longtblr}

\section{Components of the Lagrangian Gradient's Jacobian}\label{sec:components-of-the-lagrangian-gradients-jacobian}

The following table displays the non-zero components of the Jacobian.
The symbolism is the same as above.

\begin{longtblr}[

    \midrule
    \SetCell[r=2]{h} $\LagrangeFunction_{\dot\Shift_{\Normal}(n)}$ & $\LagrangeParameter_{\Dissipation}$ &
    \begin{equation*}
        -\frac{\partial\GibbsEnergy}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\
    & $\LagrangeParameter_{\Volume}(n)$ &
    \begin{equation*}
        \frac{\partial{\Volume}(n)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\

    \midrule
    \SetCell[r=4]{h} $\LagrangeFunction_{\Flux_{\Upper}(n)}$ & $\LagrangeParameter_{\Dissipation}$ &
    \begin{equation*}
        -2\frac{\GasConstant \Temperature}{\MolarVolume \VacancyConcentration^{\Standard}} \frac{\SurfaceDistance_{\Upper}(n) \Flux_{\Upper}(n)}{\DiffusionCoefficient_{\Upper}(n)}
    \end{equation*} \\
    & $\Flux_{\Upper}(n)$ &
    \begin{equation*}
        -2\frac{\GasConstant \Temperature}{\MolarVolume \VacancyConcentration^{\Standard}} \frac{\SurfaceDistance_{\Upper}(n)}{\DiffusionCoefficient_{\Upper}(n)} \LagrangeParameter_{\Dissipation}
    \end{equation*} \\
    & $\LagrangeParameter_{\Volume}(n)$ &
    \begin{equation*}
        - 1
    \end{equation*} \\
    & $\LagrangeParameter_{\Volume}(n_{\Upper})$ &
    \begin{equation*}
        + 1
    \end{equation*} \\

    \midrule
    \SetCell[r=3]{h} $\LagrangeFunction_{\LagrangeParameter_{\Volume}(n)}$ & $\dot\Shift_{\Normal}(n)$ &
    \begin{equation*}
        \frac{\partial{\Volume}(n)}{\partial\Shift_{\Normal}}
    \end{equation*} \\
    & $\Flux_{\Upper}(n)$ &
    \begin{equation*}
        - 1
    \end{equation*} \\
    & $\Flux_{\Upper}(n_{\Lower})$ &
    \begin{equation*}
        + 1
    \end{equation*} \\

    \midrule
    \SetCell[c=3]{l} for each contact node $n$ as part of contact $c$ \\
    \midrule
    \SetCell[r=2]{h} $\LagrangeFunction_{\dot\Shift_{\Normal}(n)}$        & $\LagrangeParameter_{\Dissipation}$ & same as above \\
    & $\LagrangeParameter_{\Volume}(n)$ & same as above \\
    & $\LagrangeParameter_{\Radius_{\Contact}}(n)$ &
    \begin{equation*}
        - \frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\
    & $ \LagrangeParameter_{\Angle_{\Contact}}(n)$ &
    \begin{equation*}
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\

    \midrule
    \SetCell[r=4]{h} $\LagrangeFunction_{\dot\Shift_{\Tangential}(n)}$ & $\LagrangeParameter_{\Dissipation}$ &
    \begin{equation*}
        -\frac{\partial\GibbsEnergy}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\LagrangeParameter_{\Volume}(n)$ &
    \begin{equation*}
        \frac{\partial{\Volume}(n)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\LagrangeParameter_{\Radius_{\Contact}}(n)$ &
    \begin{equation*}
        -\frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\LagrangeParameter_{\Angle_{\Contact}}(n)$ &
    \begin{equation*}
        -\frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\

    \midrule
    $\LagrangeFunction_{\Flux_{\Upper}(n)}$                                &                                 & same as above \\

    \midrule
    \SetCell[r=4]{h} $\LagrangeFunction_{\LagrangeParameter_{\Volume}(n)}$ & $\dot\Shift_{\Normal}(n)$ & same as above \\
    & $\dot\Shift_{\Tangential}(n)$ &
    \begin{equation*}
        \frac{\partial{\Volume}(n)}{\partial\Shift_{\Tangential}}
    \end{equation*} \\
    & $\Flux_{\Upper}(n)$             & same as above \\
    & $\Flux_{\Upper}(n_{\Lower})$      & same as above \\

    \midrule
    \SetCell[r=6]{h} $\LagrangeFunction_{\LagrangeParameter_{\Radius_{\Contact}}(n)}$\TblrNote{*} & $\dot\Shift_{\Normal}\Regarding\Parent(n)$ &
    \begin{equation*}
        -\frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\
    & $\dot\Shift_{\Normal}\Regarding\Child(n)$ &
    \begin{equation*}
        -\frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\
    & $\dot\Shift_{\Tangential}\Regarding\Parent(n)$ &
    \begin{equation*}
        -\frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\dot\Shift_{\Tangential}\Regarding\Child(n)$ &
    \begin{equation*}
        -\frac{\partial \Radius_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\dot\Radius_{\Contact}(c)$ &
    \begin{equation*}
        1
    \end{equation*} \\
    & $\dot\RotationAngle_{\Child}(c)$ &
    \begin{equation*}
        -\Radius\Regarding\Child(n) \sin \left( \Angle\Regarding\Child(n) - \Angle_{\Contact}(c) \right)
    \end{equation*} \\

    \midrule
    \SetCell[r=6]{h} $\LagrangeFunction_{\LagrangeParameter_{\Angle_{\Contact}}(n)}$\TblrNote{*} & $\dot\Shift_{\Normal}\Regarding\Parent(n)$ &
    \begin{equation*}
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\
    & $\dot\Shift_{\Normal}\Regarding\Child(n)$ &
    \begin{equation*}
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Normal}(n)}
    \end{equation*} \\
    & $\dot\Shift_{\Tangential}\Regarding\Parent(n)$ &
    \begin{equation*}
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\dot\Shift_{\Tangential}\Regarding\Child(n)$ &
    \begin{equation*}
        - \frac{\partial \Angle_{\Contact}(c)}{\partial\Shift_{\Tangential}(n)}
    \end{equation*} \\
    & $\dot\Angle_{\Contact}(c)$ &
    \begin{equation*}
        1
    \end{equation*} \\
    & $\dot\RotationAngle_{\Child}(c)$ &
    \begin{equation*}
        \frac{\Radius\Regarding\Child(n)}{\Radius_{\Contact}(c)} \cos \left( \Angle\Regarding\Child(n) - \Angle_{\Contact}(c) \right)
    \end{equation*} \\

    \midrule
    \SetCell[c=3]{l} for each
