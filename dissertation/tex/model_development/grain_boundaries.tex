\begin{equation}
    \frac{\partial \GibbsEnergy}{\partial \Normal{\Shift}} = \lim_{\Normal\ShiftStep \rightarrow 0} \frac{\Normal\GibbsEnergyStep}{\Normal\ShiftStep} = -2 \GrainBoundary{\InterfaceEnergy} \cos \SurfaceNormalAngle
    \label{eq:gibbs-partial-boundary-normal}
\end{equation}

\begin{equation}
    \frac{\partial \GibbsEnergy}{\partial \Tangential{\Shift}} = \lim_{\Tangential\ShiftStep \rightarrow 0} \frac{\Tangential\GibbsEnergyStep}{\Tangential\ShiftStep} = 0
    \label{eq:gibbs-partial-boundary-tangential}
\end{equation}
